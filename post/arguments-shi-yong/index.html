2、转变成数组方法:toArray()

/**
 * 取自vue2.0源码 将一个类数组对象转化成数组
 * Convert an Array-like object to a real Array.
 */
function toArray (list, start) {
  start = start || 0;
  var i = list.length - start;
  var ret = new Array(i);
  while (i--) {
    ret[i] = list[i + start];
  }
  return ret
}
